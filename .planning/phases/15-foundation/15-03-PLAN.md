<!-- Sequential failover contract (D-DEC §2.6, ARCHITECTURE §3, REQ SCRAPER-FOUND-04): -->
```
for _, p := range o.providers {
  result, err := p.<method>(ctx, args)
  if err == nil {
    return result, nil  // first success wins
  }
  // on ErrNotFound: try next (real empty from this provider, maybe another has it)
  // on ErrProviderDown: try next (this provider is sick)
  // on any other error: try next (defensive — log + metric)
  if errors.Is(err, domain.ErrProviderDown) || errors.Is(err, domain.ErrNotFound) || errors.Is(err, domain.ErrExtractFailed) {
    // log + emit parser_fallback_total{from=current, to=next}
    continue
  }
  // unrecoverable: return as-is (panic-style errors, context cancellation)
  return nil, err
}
return nil, domain.ErrNotFound  // exhausted all providers
```
