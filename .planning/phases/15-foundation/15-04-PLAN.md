<!-- Scraper-handler shape (NEW): handlers passthrough status code + body verbatim because the scraper returns 503/200 with content the catalog should not reinterpret. -->
```go
func (h *CatalogHandler) GetScraperEpisodes(w http.ResponseWriter, r *http.Request) {
    animeID := chi.URLParam(r, "animeId")
    if animeID == "" {
        httputil.BadRequest(w, "anime ID is required")
        return
    }
    prefer := r.URL.Query().Get("prefer")
    status, body, err := h.catalogService.GetScraperEpisodes(r.Context(), animeID, prefer)
    if err != nil {
        httputil.Error(w, err)
        return
    }
    // Passthrough: forward status code + body verbatim. JSON content-type assumed.
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    _, _ = w.Write(body)
}
```
