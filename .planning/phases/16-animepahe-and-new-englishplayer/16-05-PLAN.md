       ```go
       func (h *ScraperHandler) GetEpisodes(w http.ResponseWriter, r *http.Request) {
           providerID := strings.TrimSpace(r.URL.Query().Get("providerID"))
           prefer := r.URL.Query().Get("prefer")
           if providerID == "" {
               h.respondErrorWithMeta(w, http.StatusBadRequest, "INVALID_INPUT", "providerID is required", prefer)
               return
           }
           eps, err := h.svc.ListEpisodes(r.Context(), providerID, prefer)
           tried := h.svc.OrderedProviderNames(prefer)
           if err != nil {
               h.handleOrchestratorError(w, err, tried)
               return
           }
           h.respondOKWithMeta(w, map[string]any{"episodes": eps}, tried)
       }
       ```

    5. Helper functions to wrap responses with `meta.tried`:
       ```go
       func (h *ScraperHandler) respondOKWithMeta(w http.ResponseWriter, data map[string]any, tried []string) {
           if tried == nil { tried = []string{} }
           data["meta"] = map[string]any{"tried": tried}
           httputil.OK(w, data)
       }
       func (h *ScraperHandler) respondErrorWithMeta(w http.ResponseWriter, status int, code, msg string, prefer string) {
           tried := h.svc.OrderedProviderNames(prefer)
           if tried == nil { tried = []string{} }
           // Build a custom error body that includes meta.tried alongside the standard error envelope.
           w.Header().Set("Content-Type", "application/json")
           w.WriteHeader(status)
           _ = json.NewEncoder(w).Encode(map[string]any{
               "success": false,
               "error": map[string]any{"code": code, "message": msg},
               "meta":  map[string]any{"tried": tried},
           })
       }
       func (h *ScraperHandler) handleOrchestratorError(w http.ResponseWriter, err error, tried []string) {
           switch {
           case errors.Is(err, domain.ErrNotFound):
               h.respondErrorWithMetaFromChain(w, http.StatusNotFound, "NOT_FOUND", "not found", tried)
           case errors.Is(err, domain.ErrProviderDown):
               h.respondErrorWithMetaFromChain(w, http.StatusBadGateway, "PROVIDER_DOWN", err.Error(), tried)
           case errors.Is(err, domain.ErrExtractFailed):
               h.respondErrorWithMetaFromChain(w, http.StatusBadGateway, "EXTRACT_FAILED", err.Error(), tried)
           default:
               h.respondErrorWithMetaFromChain(w, http.StatusInternalServerError, "INTERNAL", err.Error(), tried)
           }
       }
       ```
