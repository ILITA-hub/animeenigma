```go
package health

import (
    "sync"
    "time"
)

const (
    failureThreshold = 3
    failureWindow    = 15 * time.Minute
)

// window is a per-(provider, stage) sliding-window failure counter.
// On every failure: prune entries older than 15min, append `now`, flip
// isDown=true if >= 3 entries remain in the window.
// On every success: reset to empty + isDown=false.
//
// Caller passes `now` explicitly (do NOT call time.Now() inside) so unit
// tests can drive the threshold deterministically (RESEARCH P-09).
type window struct {
    mu       sync.Mutex
    failures []time.Time
    isDown   bool
}

// RecordFailure adds a failure timestamp; returns true if the gauge should be 0.
func (w *window) RecordFailure(now time.Time) bool {
    w.mu.Lock()
    defer w.mu.Unlock()
    cutoff := now.Add(-failureWindow)
    pruned := w.failures[:0]
    for _, t := range w.failures {
        if t.After(cutoff) {
            pruned = append(pruned, t)
        }
    }
    w.failures = append(pruned, now)
    if len(w.failures) >= failureThreshold {
        w.isDown = true
    }
    return w.isDown
}

// RecordSuccess resets the window; returns false (gauge = 1).
func (w *window) RecordSuccess() bool {
    w.mu.Lock()
    defer w.mu.Unlock()
    w.failures = w.failures[:0]
    w.isDown = false
    return false
}

// windowSet bundles one window per (provider, stage) pair. Keyed by
// stage name; constructed once per provider in the ProbeRunner.
type windowSet struct {
    byStage map[string]*window
}

func newWindowSet() *windowSet {
    out := &windowSet{byStage: make(map[string]*window, len(AllStages))}
    for _, s := range AllStages {
        out.byStage[s] = &window{}
    }
    return out
}

func (s *windowSet) Record(stage string, now time.Time, ok bool) bool {
    w := s.byStage[stage]
    if ok {
        return w.RecordSuccess()
    }
    return w.RecordFailure(now)
}
```

Note: `RecordFailure` / `RecordSuccess` are EXPORTED (capital R) because `probe.go` in the same package can use either; the windowSet wrapper provides the dispatch helper. The lowercase `window` struct stays package-private.

```go
package health

import (
    "context"
    "errors"
    "fmt"
    "io"
    "math/rand/v2"
    "net/http"
    "os"
    "strconv"
    "time"

    "github.com/ILITA-hub/animeenigma/libs/logger"
    "github.com/ILITA-hub/animeenigma/libs/metrics"
    "github.com/ILITA-hub/animeenigma/services/scraper/internal/domain"
)

const (
    probeBaseInterval = 15 * time.Minute
    probeJitterPct    = 0.20
    segmentTimeout    = 10 * time.Second
)

// ProbeRunner is a per-provider liveness probe goroutine. One instance per
// registered provider, started from main.go after Orchestrator.Register().
//
// Per tick:
//   1. Pick a random AnimeRef from the golden pool.
//   2. Call provider.FindID → record (search) success/failure.
//   3. On success: provider.ListEpisodes → record (episodes).
//   4. On success: provider.ListServers → record (servers).
//   5. On success: provider.GetStream → record (stream).
//   6. On success: HTTP GET first 4KB of Sources[0].URL → record (stream_segment).
//   7. Update cache with the per-stage StageStatus map.
//   8. Update gauges from the windowSet's current up/down state.
//   9. Heartbeat: metrics.ProviderProbeLastTick.WithLabelValues(name).Set(now.Unix()).
//
// On any failure: short-circuit — do NOT exercise subsequent stages this tick.
type ProbeRunner struct {
    provider domain.Provider
    pool     []domain.AnimeRef
    cache    *InMemoryHealthCache
    log      *logger.Logger
    windows  *windowSet
    http     *http.Client            // for the segment-fetch stage
    now      func() time.Time
    rng      *rand.Rand
}

type ProbeOption func(*ProbeRunner)

func WithNow(fn func() time.Time) ProbeOption { return func(r *ProbeRunner) { r.now = fn } }
func WithRNG(rng *rand.Rand) ProbeOption       { return func(r *ProbeRunner) { r.rng = rng } }
func WithHTTPClient(c *http.Client) ProbeOption { return func(r *ProbeRunner) { r.http = c } }

func NewProbeRunner(p domain.Provider, pool []domain.AnimeRef, cache *InMemoryHealthCache, log *logger.Logger, opts ...ProbeOption) *ProbeRunner {
    r := &ProbeRunner{
        provider: p,
        pool:     pool,
        cache:    cache,
        log:      log,
        windows:  newWindowSet(),
        http:     &http.Client{Timeout: segmentTimeout},
        now:      time.Now,
        rng:      rand.New(rand.NewPCG(uint64(time.Now().UnixNano()), 0)),
    }
    for _, opt := range opts {
        opt(r)
    }
    return r
}

// Start blocks until ctx is cancelled. Designed for `go r.Start(ctx)`.
// Wraps the per-tick loop in defer-recover; on panic, logs + re-spawns
// (RESEARCH P-07).
func (r *ProbeRunner) Start(ctx context.Context) {
    defer func() {
        if rec := recover(); rec != nil {
            r.log.Errorw("scraper.probe: panicked, restarting",
                "provider", r.provider.Name(),
                "panic", rec,
            )
            // Re-spawn so a single bad tick doesn't kill observability.
            go r.Start(ctx)
        }
    }()
    r.log.Infow("scraper.probe: started",
        "provider", r.provider.Name(),
        "pool_size", len(r.pool),
        "base_interval", probeBaseInterval.String(),
    )

    // First tick after randomized initial delay (RESEARCH P-06 — cookie-jar
    // warmup race + boot stampede prevention). Range: 0 to interval/4.
    //
    // Test/CI fast-path: if SCRAPER_PROBE_INITIAL_DELAY_OVERRIDE_SECONDS is
    // set to a non-empty integer, use that fixed delay instead of the random
    // jittered one. This lets the verification step in Task 4 land the first
    // tick in ~5s rather than waiting up to 3.75 min in production-default
    // mode. Leave unset (empty) for normal jittered behavior.
    var initialDelay time.Duration
    if envVal := os.Getenv("SCRAPER_PROBE_INITIAL_DELAY_OVERRIDE_SECONDS"); envVal != "" {
        if secs, err := strconv.Atoi(envVal); err == nil && secs >= 0 {

                "provider", r.provider.Name(),
                "delay_seconds", secs,
            )
        } else {
            initialDelay = time.Duration(r.rng.Int64N(int64(probeBaseInterval / 4)))
        }
    } else {
        initialDelay = time.Duration(r.rng.Int64N(int64(probeBaseInterval / 4)))
    }
    select {
    case <-ctx.Done():
        return
    case <-time.After(initialDelay):
    }

    for {
        r.runOneTick(ctx)
        sleep := nextSleep(r.rng)
        select {
        case <-ctx.Done():
            r.log.Infow("scraper.probe: stopped", "provider", r.provider.Name())
            return
        case <-time.After(sleep):
        }
    }
}

// nextSleep returns probeBaseInterval ± probeJitterPct.
func nextSleep(rng *rand.Rand) time.Duration {
    delta := (rng.Float64()*2 - 1) * probeJitterPct
    return time.Duration(float64(probeBaseInterval) * (1 + delta))
}

// RunOnce exercises one tick. Exposed for tests; production caller uses Start.
func (r *ProbeRunner) RunOnce(ctx context.Context) {
    r.runOneTick(ctx)
}

func (r *ProbeRunner) runOneTick(ctx context.Context) {
    name := r.provider.Name()
    ref := Pick(r.pool, r.rng)
    stages := make(map[string]StageStatus, len(AllStages))
    now := r.now()

    // Stage 1: search (FindID)
    providerID, err := r.provider.FindID(ctx, ref)
    if !r.record(stages, StageSearch, now, err) {
        r.commit(name, stages, now)
        return
    }

    // Stage 2: episodes
    episodes, err := r.provider.ListEpisodes(ctx, providerID)
    if err == nil && len(episodes) == 0 {
        err = fmt.Errorf("list_episodes returned 0 items")
    }
    if !r.record(stages, StageEpisodes, now, err) {
        r.commit(name, stages, now)
        return
    }

    // Stage 3: servers
    servers, err := r.provider.ListServers(ctx, providerID, episodes[0].ID)
    if err == nil && len(servers) == 0 {
        err = fmt.Errorf("list_servers returned 0 items")
    }
    if !r.record(stages, StageServers, now, err) {
        r.commit(name, stages, now)
        return
    }

    // Stage 4: stream
    stream, err := r.provider.GetStream(ctx, providerID, episodes[0].ID, servers[0].ID, domain.CategorySub)
    if err == nil && (stream == nil || len(stream.Sources) == 0) {
        err = fmt.Errorf("get_stream returned no sources")
    }
    if !r.record(stages, StageStream, now, err) {
        r.commit(name, stages, now)
        return
    }

    // Stage 5: stream_segment — HEAD or short GET on Sources[0].URL.
    segErr := r.fetchSegment(ctx, stream.Sources[0].URL)
    r.record(stages, StageStreamSegment, now, segErr)
    r.commit(name, stages, now)
}

// record updates a window + the cache's stage entry. Returns whether the
// stage succeeded (caller short-circuits if false).
// Truncates the error message to MaxLastErrChars before storing (P-05).
func (r *ProbeRunner) record(stages map[string]StageStatus, stage string, now time.Time, err error) bool {
    ok := (err == nil)
    isDown := r.windows.Record(stage, now, ok)
    var lastErr string
    if err != nil {
        msg := err.Error()
        if len(msg) > MaxLastErrChars {
            msg = msg[:MaxLastErrChars]
        }
        lastErr = msg
    }
    s := StageStatus{
        Up:      !isDown,  // gauge follows window, not raw stage result
        LastErr: lastErr,
    }
    if ok {
        s.LastOK = now
    } else if prev, hadPrev := stages[stage]; hadPrev {
        // Preserve prior LastOK; tests can pre-populate via cache.
        s.LastOK = prev.LastOK
    }
    stages[stage] = s
    return ok
}

// commit writes the per-stage map to the cache + gauges + heartbeat.
func (r *ProbeRunner) commit(name string, stages map[string]StageStatus, now time.Time) {
    r.cache.Update(name, ProviderHealth{
        Stages:      stages,
        LastUpdated: now,
    })
    // Emit gauges for ALL canonical stages (including ones not exercised
    // this tick — they keep their last-known value via the windowSet state).
    for _, s := range AllStages {
        up := 1.0
        if status, ok := stages[s]; ok && !status.Up {
            up = 0.0
        }
        metrics.ProviderHealthUp.WithLabelValues(name, s).Set(up)
    }
    // Heartbeat
    metrics.ProviderProbeLastTick.WithLabelValues(name).Set(float64(now.Unix()))
}

// fetchSegment issues a bounded GET of the first 4KB of the source URL.
// Counts as success only if HTTP 200 + at least one non-empty byte.
func (r *ProbeRunner) fetchSegment(ctx context.Context, urlStr string) error {
    if urlStr == "" {
        return errors.New("stream_segment: empty source URL")
    }
    ctx, cancel := context.WithTimeout(ctx, segmentTimeout)
    defer cancel()
    req, err := http.NewRequestWithContext(ctx, http.MethodGet, urlStr, nil)
    if err != nil {
        return fmt.Errorf("stream_segment: build request: %w", err)
    }
    resp, err := r.http.Do(req)
    if err != nil {
        return fmt.Errorf("stream_segment: do request: %w", err)
    }
    defer resp.Body.Close()
    if resp.StatusCode/100 != 2 {
        return fmt.Errorf("stream_segment: status %d", resp.StatusCode)
    }
    buf := make([]byte, 4096)
    n, _ := io.ReadFull(resp.Body, buf)
    if n == 0 {
        return errors.New("stream_segment: empty body")
    }
    return nil
}
```
