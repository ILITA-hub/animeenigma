1. **VibePlayerExtractor** (`embeds/vibeplayer.go`) — regex-only. Fetches the wrapper page, extracts `const src = "https://vibeplayer.site/public/stream/<id>/master.m3u8"` and optional `const subtitle = "..."` via two compiled regexes. No goja. Allowlist: `["vibeplayer.site"]`.

Allowlist: `["otakuhg.site"]`. Referer for upstream: `https://otakuhg.site/`.

    ```go
    package embeds

    import (
        "context"
        "errors"
        "fmt"
        "io"
        "net/http"
        "net/url"
        "regexp"
        "strings"
        "time"

        "github.com/ILITA-hub/animeenigma/libs/metrics"
        "github.com/ILITA-hub/animeenigma/services/scraper/internal/domain"
    )

    const defaultPackedHTTPTimeout = 15 * time.Second
    const maxPackedBody = 2 << 20

    // packedExtractor is the shared Dean-Edwards-packer extractor used by
    // StreamHG and Earnvids. Both providers' wrapper HTML matches the kwik
    // unpacker shape but emits the HLS URL under a different field name
    // (`hls2` instead of `source`). One packer; two consumers.
    //
    // To avoid duplicating the goja runtime + brace balancer, this type lives
    // in the same package as KwikExtractor and reuses its unexported helpers.
    type packedExtractor struct {
        name               string
        hosts              []string
        referer            string
        selectorPackerFail string
        selectorRegexFail  string
        selectorBodyFail   string
        http               *http.Client
        timeout            time.Duration
    }

    // hls2Regex finds the `"hls2":"<url>"` field inside the unpacked packer body.
    // Distinct from kwik.go's regex (kwik emits `source = "..."` directly).
    var hls2Regex = regexp.MustCompile(`"hls2"\s*:\s*"(https://[^"]+\.m3u8[^"]*)"`)

    func (e *packedExtractor) Name() string { return e.name }

    func (e *packedExtractor) Matches(embedURL string) bool {
        u, err := url.Parse(embedURL)
        if err != nil { return false }
        if u.Scheme != "http" && u.Scheme != "https" { return false }
        host := strings.ToLower(u.Hostname())
        if host == "" { return false }
        for _, known := range e.hosts {
            if host == known || strings.HasSuffix(host, "."+known) { return true }
        }
        return false
    }

    func (e *packedExtractor) Extract(ctx context.Context, embedURL string, headers http.Header) (*domain.Stream, error) {
        if !e.Matches(embedURL) {
            return nil, domain.WrapExtractFailed(
                fmt.Errorf("host not in allowlist"),
                e.name+": Matches gate")
        }
        req, err := http.NewRequestWithContext(ctx, http.MethodGet, embedURL, nil)
        if err != nil {
            return nil, domain.WrapProviderDown(err, e.name+": build request")
        }
        // Caller-supplied headers first (so callers can override UA, Accept, etc.).
        for k, vs := range headers { req.Header[k] = vs }
        // Force Referer to the wrapper origin if the caller didn't set one.
        if req.Header.Get("Referer") == "" { req.Header.Set("Referer", e.referer) }

        resp, err := e.http.Do(req)
        if err != nil { return nil, domain.WrapProviderDown(err, e.name+": fetch") }
        defer func() {
            _, _ = io.Copy(io.Discard, resp.Body)
            _ = resp.Body.Close()
        }()
        if resp.StatusCode >= 400 {
            return nil, domain.WrapProviderDown(
                fmt.Errorf("upstream %d", resp.StatusCode),
                e.name+": HTTP status")
        }
        body, err := io.ReadAll(io.LimitReader(resp.Body, maxPackedBody))
        if err != nil {
            metrics.ParserZeroMatchTotal.WithLabelValues(e.name, e.selectorBodyFail).Inc()
            return nil, domain.WrapProviderDown(err, e.name+": read")
        }

        // Reuse kwik.go's packer-extraction helpers (same package).
        packerIIFE, ok := findPackerIIFE(body)  // local helper, see below
        if !ok {
            metrics.ParserZeroMatchTotal.WithLabelValues(e.name, e.selectorPackerFail).Inc()
            return nil, domain.WrapExtractFailed(errors.New("no packer IIFE"), e.name+": packer locate")
        }

        unpacked, err := runPackerGoja(ctx, packerIIFE, e.timeout)
        if err != nil {
            metrics.ParserZeroMatchTotal.WithLabelValues(e.name, e.selectorPackerFail).Inc()
            return nil, domain.WrapExtractFailed(err, e.name+": goja unpack")
        }

        m := hls2Regex.FindStringSubmatch(unpacked)
        if m == nil {
            metrics.ParserZeroMatchTotal.WithLabelValues(e.name, e.selectorRegexFail).Inc()
            return nil, domain.WrapExtractFailed(errors.New("no hls2 in unpacked body"), e.name+": hls2 regex")
        }

        return &domain.Stream{
            Sources: []domain.Source{{URL: m[1], Type: "hls"}},
            Headers: map[string]string{"Referer": e.referer},
        }, nil
    }
    ```

    `findPackerIIFE(body []byte) (string, bool)` — implement (or wire to existing kwik.go helpers). If kwik.go exposes a working `extractPacker(body)` or similar helper, call it directly. Otherwise:
    ```go
    var packerStartPattern = regexp.MustCompile(`eval\(function\(p,a,c,k,e,d\)`)

    func findPackerIIFE(body []byte) (string, bool) {
        loc := packerStartPattern.FindIndex(body)
        if loc == nil { return "", false }
        start := loc[0]
        // Balance parens starting at the opening '(' after `eval`.
        // Find the first '(' at or after loc[0]+4 (length of "eval").
        i := start + 4
        for i < len(body) && body[i] != '(' { i++ }
        if i >= len(body) { return "", false }
        depth := 0
        end := -1
        for j := i; j < len(body); j++ {
            switch body[j] {
            case '(': depth++
            case ')': depth--
                if depth == 0 { end = j + 1; break }
            }
            if end >= 0 { break }
        }
        if end < 0 { return "", false }
        return string(body[start:end]), true
    }
    ```

    ```go
    import "github.com/dop251/goja"

    func runPackerGoja(ctx context.Context, expr string, timeout time.Duration) (string, error) {
        vm := goja.New()
        done := make(chan struct{})
        defer close(done)

        go func() {
            timer := time.NewTimer(timeout)
            defer timer.Stop()
            select {
            case <-timer.C:    vm.Interrupt("packed: unpack timeout")
            case <-ctx.Done(): vm.Interrupt("packed: ctx cancel")
            case <-done:
            }
        }()

        val, err := vm.RunString(expr)
        if err != nil { return "", err }
        if val == nil || goja.IsUndefined(val) || goja.IsNull(val) {
            return "", errors.New("packed: goja returned nil/undefined")
        }
        return val.String(), nil
    }
    ```

    ```go
    package embeds

    import (
        "context"
        "errors"
        "fmt"
        "io"
        "net/http"
        "net/url"
        "regexp"
        "strings"
        "time"

        "github.com/ILITA-hub/animeenigma/libs/metrics"
        "github.com/ILITA-hub/animeenigma/services/scraper/internal/domain"
    )

    const (
        defaultVibePlayerHTTPTimeout = 15 * time.Second
        maxVibePlayerBody            = 2 << 20

        selectorVibePlayerSrcRegex = "vibeplayer_src_const"
        selectorVibePlayerBodyRead = "vibeplayer_body_read"
    )

    var vibePlayerHosts = []string{"vibeplayer.site"}

    var (
        vibePlayerSrcRegex = regexp.MustCompile(`const\s+src\s*=\s*"(https://[^"]+\.m3u8)"`)
        vibePlayerSubRegex = regexp.MustCompile(`const\s+subtitle\s*=\s*"([^"]*)"`)
    )

    // VibePlayerExtractor pulls the inline `const src="...m3u8"` URL out of a
    // vibeplayer.site wrapper page. No goja — pure regex.
    type VibePlayerExtractor struct {
        http    *http.Client
        timeout time.Duration
    }

    func NewVibePlayerExtractor() *VibePlayerExtractor {
        return &VibePlayerExtractor{
            http:    &http.Client{Timeout: defaultVibePlayerHTTPTimeout},
            timeout: defaultVibePlayerHTTPTimeout,
        }
    }

    func (e *VibePlayerExtractor) Name() string { return "vibeplayer" }

    func (e *VibePlayerExtractor) Matches(embedURL string) bool {
        u, err := url.Parse(embedURL)
        if err != nil { return false }
        if u.Scheme != "http" && u.Scheme != "https" { return false }
        host := strings.ToLower(u.Hostname())
        if host == "" { return false }
        for _, known := range vibePlayerHosts {
            if host == known || strings.HasSuffix(host, "."+known) { return true }
        }
        return false
    }

    func (e *VibePlayerExtractor) Extract(ctx context.Context, embedURL string, headers http.Header) (*domain.Stream, error) {
        if !e.Matches(embedURL) {
            return nil, domain.WrapExtractFailed(
                errors.New("host not allowlisted"),
                "vibeplayer: Matches gate")
        }
        req, err := http.NewRequestWithContext(ctx, http.MethodGet, embedURL, nil)
        if err != nil { return nil, domain.WrapProviderDown(err, "vibeplayer: build request") }
        for k, vs := range headers { req.Header[k] = vs }
        if req.Header.Get("Referer") == "" {
            req.Header.Set("Referer", "https://anitaku.to/")
        }
        resp, err := e.http.Do(req)
        if err != nil { return nil, domain.WrapProviderDown(err, "vibeplayer: fetch") }
        defer func() {
            _, _ = io.Copy(io.Discard, resp.Body)
            _ = resp.Body.Close()
        }()
        if resp.StatusCode >= 400 {
            return nil, domain.WrapProviderDown(
                fmt.Errorf("upstream %d", resp.StatusCode), "vibeplayer: HTTP status")
        }
        body, err := io.ReadAll(io.LimitReader(resp.Body, maxVibePlayerBody))
        if err != nil {
            metrics.ParserZeroMatchTotal.WithLabelValues("vibeplayer", selectorVibePlayerBodyRead).Inc()
            return nil, domain.WrapProviderDown(err, "vibeplayer: read")
        }
        srcM := vibePlayerSrcRegex.FindSubmatch(body)
        if srcM == nil {
            metrics.ParserZeroMatchTotal.WithLabelValues("vibeplayer", selectorVibePlayerSrcRegex).Inc()
            return nil, domain.WrapExtractFailed(errors.New("no src= const"), "vibeplayer: src regex")
        }
        stream := &domain.Stream{
            Sources: []domain.Source{{URL: string(srcM[1]), Type: "hls"}},
            Headers: map[string]string{"Referer": "https://vibeplayer.site/"},
        }
        if subM := vibePlayerSubRegex.FindSubmatch(body); subM != nil && len(subM[1]) > 0 {
            stream.Tracks = []domain.Track{
                {File: string(subM[1]), Label: "English", Kind: "captions", Default: true},
            }
        }
        return stream, nil
    }

    var _ domain.EmbedExtractor = (*VibePlayerExtractor)(nil)
    ```
