<objective>
Replace the Plan-01 Aggregator.Resolve stub with the concurrent fan-out
implementation: each resolver runs in its own goroutine under a per-card
`context.WithTimeout(ctx, 800ms)`; the parent context carries a 2s overall
budget; failed/timed-out resolvers drop their card and emit one
`spotlight.card_failed` Errorw log; resolvers that return `(nil, nil)`
silently drop without a log; the final card list is shuffled-but-no, just
appended in completion order (frontend handles ordering); on a zero-card
outcome the aggregator attempts a snapshot fallback via Redis; on a
non-zero outcome it writes the snapshot best-effort via a detached
`context.Background()` goroutine.

Purpose: This plan ships the "drop-on-error, partial-success" semantics
that the entire HSB-BE-03/04/05 requirement set hinges on. Pattern lifted
from `services/catalog/internal/service/subs_aggregator.go` lines 109-156
(buffered chan + WaitGroup + drop-on-error) with three deliberate
adaptations: (1) per-card ctx.WithTimeout instead of trusting the outer
ctx; (2) `(nil, nil)` is a silent-drop, not a log; (3) detached snapshot
write because the request ctx will be cancelled by the time the snapshot
write begins.

       ```
       func (a *Aggregator) Resolve(ctx context.Context, userID *string) (*Response, error) {
           // Overall 2s budget — HSB-BE-04. Wraps the caller's ctx so a
           // long-running resolver cannot outlive the request.
           ctx, cancel := context.WithTimeout(ctx, overallBudget)
           defer cancel()

           if len(a.resolvers) == 0 {
               // No resolvers wired (test path) — try snapshot, else empty.
               if snap := a.loadSnapshot(ctx, userID); snap != nil {
                   return snap, nil
               }
               return &Response{Cards: []Card{}, GeneratedAt: time.Now().UTC().Format(time.RFC3339)}, nil
           }

           resultsCh := make(chan resolveResult, len(a.resolvers))
           var wg sync.WaitGroup

           for _, r := range a.resolvers {
               wg.Add(1)
               go func(r Resolver) {
                   defer wg.Done()
                   cctx, cancel := context.WithTimeout(ctx, perCardDeadline)
                   defer cancel()
                   card, err := r.Resolve(cctx, userID)
                   resultsCh <- resolveResult{name: r.Type(), card: card, err: err}
               }(r)
           }

           go func() { wg.Wait(); close(resultsCh) }()

           cards := []Card{}
           for res := range resultsCh {
               if res.err != nil {
                   // Per HSB-BE-03: dropped card gets ONE structured log line.
                   a.log.Errorw("spotlight.card_failed", "type", res.name, "error", res.err)
                   continue
               }
               if res.card == nil {
                   // Eligible=false — silent drop, no log (Plan 01 contract).
                   continue
               }
               cards = append(cards, *res.card)
           }

           // Zero-card outcome — try snapshot fallback (HSB-BE-04).
           if len(cards) == 0 {
               if snap := a.loadSnapshot(ctx, userID); snap != nil {
                   return snap, nil
               }
           }

           resp := &Response{
               Cards:       cards,
               GeneratedAt: time.Now().UTC().Format(time.RFC3339),
           }

           // Best-effort snapshot write — detached ctx (request ctx is about
           // to be cancelled by the caller). Fire-and-forget; never blocks
           // the response path. Only write on ≥1 card to avoid baking empty
           // results into a 24h snapshot.
           if len(cards) > 0 {
               go a.saveSnapshot(context.Background(), userID, resp)
           }

           return resp, nil
       }
       ```

    5. Private helpers (file-local):
       ```
       // loadSnapshot returns the cached fallback Response for the given
       // userID, or nil if none exists / Redis is down. ErrNotFound is the
       // only error path that does NOT emit a log; other errors get a Warnw
       // and return nil (request continues with whatever cards resolved).
       func (a *Aggregator) loadSnapshot(ctx context.Context, userID *string) *Response {
           var snap Response
           err := a.cache.Get(ctx, SnapshotKey(userID), &snap)
           if err == nil {
               return &snap
           }
           if !errors.Is(err, cache.ErrNotFound) {
               a.log.Warnw("spotlight.snapshot_load_failed", "error", err)
           }
           return nil
       }

       // saveSnapshot writes the best-known-good Response to Redis under the
       // per-day snapshot key. Detached ctx so request cancellation does
       // not abort the write. Errors are Warnw and dropped.
       func (a *Aggregator) saveSnapshot(ctx context.Context, userID *string, resp *Response) {
           if err := a.cache.Set(ctx, SnapshotKey(userID), resp, snapshotTTL); err != nil {
               a.log.Warnw("spotlight.snapshot_save_failed", "error", err)
           }
       }
       ```
