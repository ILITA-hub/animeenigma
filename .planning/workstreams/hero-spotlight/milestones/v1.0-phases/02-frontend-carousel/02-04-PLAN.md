```typescript
<script setup lang="ts">
import { computed, ref, watch } from 'vue'
import { useI18n } from 'vue-i18n'
import { useIntervalFn, useMediaQuery } from '@vueuse/core'
import { useSpotlight } from '@/composables/useSpotlight'
import type { SpotlightCard } from '@/types/spotlight'
import CarouselControls from './CarouselControls.vue'
import AnimeOfDayCard from './cards/AnimeOfDayCard.vue'
import RandomTailCard from './cards/RandomTailCard.vue'
import LatestNewsCard from './cards/LatestNewsCard.vue'
import PlatformStatsCard from './cards/PlatformStatsCard.vue'
import { getLocalizedTitle } from '@/utils/title'

const AUTO_CYCLE_INTERVAL_MS = 7000

const enabled =
  (import.meta.env.VITE_HERO_SPOTLIGHT_ENABLED as string | undefined) !== 'false'

const { t } = useI18n()
const { cards, loading } = useSpotlight()
const reducedMotion = useMediaQuery('(prefers-reduced-motion: reduce)')

const rootRef = ref<HTMLElement | null>(null)
const currentIndex = ref(0)

function advance() {
  if (cards.value.length === 0) return
  currentIndex.value = (currentIndex.value + 1) % cards.value.length
}

const { pause, resume, isActive } = useIntervalFn(advance, AUTO_CYCLE_INTERVAL_MS, {
  immediate: false,
})

function startCycle() {
  if (reducedMotion.value) return
  if (cards.value.length <= 1) return
  resume()
}

function stopCycle() { pause() }

function restart() {
  pause()
  startCycle()
}

function next() {
  if (cards.value.length === 0) return
  currentIndex.value = (currentIndex.value + 1) % cards.value.length
  restart()
}

function prev() {
  if (cards.value.length === 0) return
  currentIndex.value = (currentIndex.value - 1 + cards.value.length) % cards.value.length
  restart()
}

function goTo(i: number) {
  currentIndex.value = i
  restart()
}

// Pitfall 4 mitigation — randomize AFTER cards populate (fetch is async)
watch(
  () => cards.value.length,
  (n) => {
    if (n > 0) {
      currentIndex.value = Math.floor(Math.random() * n)
      startCycle()
    }
  },
  { immediate: false },
)

// Pitfall 7 mitigation — focusout fires before activeElement settles
function onFocusOut() {
  // Defer to next frame so document.activeElement is stable
  requestAnimationFrame(() => {
    if (!rootRef.value?.contains(document.activeElement)) {
      startCycle()
    }
  })
}

// Reduced-motion runtime toggle: if OS setting changes, pause immediately
watch(reducedMotion, (v) => {
  if (v) pause()
  else if (cards.value.length > 1) resume()
})

const active = computed<SpotlightCard | null>(() =>
  cards.value.length > 0 ? cards.value[currentIndex.value] ?? null : null,
)

const CARD_MAP = {
  anime_of_day: AnimeOfDayCard,
  random_tail: RandomTailCard,
  latest_news: LatestNewsCard,
  platform_stats: PlatformStatsCard,
} as const

function cardFor(type: SpotlightCard['type']) {
  return CARD_MAP[type]
}

function cardTitle(card: SpotlightCard): string {
  switch (card.type) {
    case 'anime_of_day':
    case 'random_tail':
      return getLocalizedTitle(
        card.data.anime.name,
        card.data.anime.name_ru,
        card.data.anime.name_jp,
      )
    case 'latest_news':
      return t('spotlight.latestNews.title')
    case 'platform_stats':
      return t('spotlight.platformStats.title')
  }
}
</script>
```
