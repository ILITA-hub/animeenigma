```typescript
import { test, expect } from '@playwright/test'
import AxeBuilder from '@axe-core/playwright'

const SPOTLIGHT_SELECTOR = 'section[role="region"][aria-roledescription="carousel"]'
const SLIDE_SELECTOR = `${SPOTLIGHT_SELECTOR} [role="group"][aria-roledescription="slide"]`
const DOT_SELECTOR = `${SPOTLIGHT_SELECTOR} [role="tablist"] button`
const NEXT_CHEVRON_SELECTOR = `${SPOTLIGHT_SELECTOR} button[aria-label*="Next"], ${SPOTLIGHT_SELECTOR} button[aria-label*="Следующий"]`
const PREV_CHEVRON_SELECTOR = `${SPOTLIGHT_SELECTOR} button[aria-label*="Previous"], ${SPOTLIGHT_SELECTOR} button[aria-label*="Предыдущий"]`
const TRENDING_RECS_TEXT_PROBE = 'home.trendingRow'  // i18n key precedent; or fall back to a known row title

test.describe('hero spotlight block (Phase 2)', () => {
  test.beforeEach(async ({ page }) => {
    // Anonymous user — Phase 2 cards are all anon-eligible
    await page.goto('/')
    // Wait for spotlight either to mount (cards loaded) or to remain hidden (empty/error)
    await page.waitForLoadState('networkidle')
  })

  test('mounts above the legacy trendingRecs row (additive Phase 2)', async ({ page }) => {
    // Block is mounted (when backend returns ≥1 card)
    const block = page.locator(SPOTLIGHT_SELECTOR)
    // The block MAY be hidden if backend returns 0 cards (e.g. all resolvers failed).
    // For Phase 2 the demo expectation is that 4 cards return. Skip gracefully if empty.
    const blockVisible = await block.count() > 0
    if (!blockVisible) {
      test.skip(true, 'Spotlight returned 0 cards — backend issue, not a frontend bug')
    }

    // The trendingRecs row should still exist (Phase 3 removes it — not Phase 2)
    // Use a structural probe: the trendingRecs container has v-if="trendingRecs.length > 0"
    // so it may or may not be visible depending on the recommendation data.
    // What we verify: the spotlight section appears BEFORE the search bar's section in DOM order
    const blockBox = await block.boundingBox()
    const searchBar = page.locator('input[placeholder]').first()
    const searchBox = await searchBar.boundingBox()
    if (blockBox && searchBox) {
      expect(blockBox.y).toBeLessThan(searchBox.y)
    }
  })

  test('auto-cycles every 7 seconds', async ({ page }) => {
    const block = page.locator(SPOTLIGHT_SELECTOR)
    await expect(block).toBeVisible()
    const slide = block.locator('[role="group"][aria-roledescription="slide"]')
    const initialLabel = await slide.getAttribute('aria-label')
    expect(initialLabel).toBeTruthy()

    // Move mouse OUT of block so hover-pause is inactive
    await page.mouse.move(0, 0)
    // Wait 7.5s — auto-cycle should fire once
    await page.waitForTimeout(7500)
    const nextLabel = await slide.getAttribute('aria-label')
    expect(nextLabel).not.toBe(initialLabel)
  })

  test('pauses on hover', async ({ page }) => {
    const block = page.locator(SPOTLIGHT_SELECTOR)
    await expect(block).toBeVisible()
    const slide = block.locator('[role="group"][aria-roledescription="slide"]')
    // Hover the block to pause auto-cycle
    await block.hover()
    const initialLabel = await slide.getAttribute('aria-label')
    await page.waitForTimeout(7500)
    const sameLabel = await slide.getAttribute('aria-label')
    expect(sameLabel).toBe(initialLabel)
  })

  test('ArrowRight key seeks to next slide', async ({ page }) => {
    const block = page.locator(SPOTLIGHT_SELECTOR)
    await expect(block).toBeVisible()
    await block.focus()
    const slide = block.locator('[role="group"][aria-roledescription="slide"]')
    const initialLabel = await slide.getAttribute('aria-label')
    await page.keyboard.press('ArrowRight')
    // Slight wait for re-render
    await page.waitForTimeout(500)
    const nextLabel = await slide.getAttribute('aria-label')
    expect(nextLabel).not.toBe(initialLabel)
  })

  test('reduced-motion preference disables auto-cycle', async ({ browser }) => {
    const context = await browser.newContext({ reducedMotion: 'reduce' })
    const page = await context.newPage()
    await page.goto('/')
    await page.waitForLoadState('networkidle')
    const block = page.locator(SPOTLIGHT_SELECTOR)
    if (await block.count() === 0) {
      test.skip(true, 'Spotlight returned 0 cards')
    }
    const slide = block.locator('[role="group"][aria-roledescription="slide"]')
    await page.mouse.move(0, 0)
    const initialLabel = await slide.getAttribute('aria-label')
    await page.waitForTimeout(8000)
    const sameLabel = await slide.getAttribute('aria-label')
    expect(sameLabel).toBe(initialLabel)
    // But manual nav still works
    await block.focus()
    await page.keyboard.press('ArrowRight')
    await page.waitForTimeout(500)
    const afterArrowLabel = await slide.getAttribute('aria-label')
    expect(afterArrowLabel).not.toBe(initialLabel)
    await context.close()
  })

  test('mobile viewport stacks card content vertically', async ({ browser }) => {
    const context = await browser.newContext({ viewport: { width: 375, height: 667 } })
    const page = await context.newPage()
    await page.goto('/')
    await page.waitForLoadState('networkidle')
    const block = page.locator(SPOTLIGHT_SELECTOR)
    if (await block.count() === 0) {
      test.skip(true, 'Spotlight returned 0 cards')
    }
    await expect(block).toBeVisible()
    // Block should respect mobile min-height (400px)
    const box = await block.boundingBox()
    expect(box?.height).toBeGreaterThanOrEqual(400)
    await context.close()
  })

  test('axe-core reports zero a11y violations on the block', async ({ page }) => {
    const block = page.locator(SPOTLIGHT_SELECTOR)
    if (await block.count() === 0) {
      test.skip(true, 'Spotlight returned 0 cards')
    }
    await expect(block).toBeVisible()
    const results = await new AxeBuilder({ page })
      .include(SPOTLIGHT_SELECTOR)
      .analyze()
    expect(results.violations, JSON.stringify(results.violations, null, 2)).toEqual([])
  })

  test('renders dot indicators equal to card count', async ({ page }) => {
    const block = page.locator(SPOTLIGHT_SELECTOR)
    if (await block.count() === 0) {
      test.skip(true, 'Spotlight returned 0 cards')
    }
    const dots = block.locator('[role="tablist"] button')
    const dotCount = await dots.count()
    expect(dotCount).toBeGreaterThanOrEqual(1)
    expect(dotCount).toBeLessThanOrEqual(4) // 4 static cards in Phase 2
  })

  test('dot click jumps to that slide (active aria-current=true)', async ({ page }) => {
    const block = page.locator(SPOTLIGHT_SELECTOR)
    if (await block.count() === 0) {
      test.skip(true, 'Spotlight returned 0 cards')
    }
    const dots = block.locator('[role="tablist"] button')
    const count = await dots.count()
    if (count < 2) {
      test.skip(true, 'Need at least 2 cards to test dot navigation')
    }
    // Pause auto-cycle first
    await block.hover()
    // Click dot index 1 (second dot)
    await dots.nth(1).click()
    await page.waitForTimeout(500)
    await expect(dots.nth(1)).toHaveAttribute('aria-current', 'true')
    // Other dots should be false
    await expect(dots.nth(0)).toHaveAttribute('aria-current', 'false')
  })
})

test.describe('hero spotlight block — flag off', () => {
  // This test is parameterized by a separate web-server start with
  // VITE_HERO_SPOTLIGHT_ENABLED=false. The simpler approach in CI is a
  // SEPARATE test run; here we mark this `test.skip` by default and provide
  // the command in the SUMMARY for manual verification.
  test.skip('block does not mount when VITE_HERO_SPOTLIGHT_ENABLED=false', async ({ page }) => {
    await page.goto('/')
    const block = page.locator('section[role="region"][aria-roledescription="carousel"]')
    await expect(block).toHaveCount(0)
    // trendingRecs row should still be reachable (if backend has recs)
  })
})
```

Notes on test pragmatism:
- The 9 ROADMAP success criteria #5 "random initial slide ~75% in 10 reloads" is hard to test deterministically in CI. The auto-cycle test (#3) is the more robust proxy; the random-init claim is covered by Vitest in Plan 04 (30-mount sample). Add a comment in the spec noting this rationale.
- Success criterion #10 (flag-off) requires a separate dev-server invocation with VITE_HERO_SPOTLIGHT_ENABLED=false; this is handled via a `test.skip()` + documented manual command in the SUMMARY. Do NOT block the phase on automating the rebuild.
- Use `test.skip(true, '<reason>')` (the playwright-correct conditional skip API) so the test marks as skipped rather than failing when the backend returns 0 cards (which happens if Phase 1 service is down or in flag-off mode). This makes the suite resilient to environmental issues while still failing hard on real frontend bugs.
