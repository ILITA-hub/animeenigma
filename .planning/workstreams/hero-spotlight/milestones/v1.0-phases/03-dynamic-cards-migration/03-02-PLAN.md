```go
func OptionalAuthMiddleware(jwtConfig authz.JWTConfig) func(http.Handler) http.Handler {
    jwtManager := authz.NewJWTManager(jwtConfig)
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            token := httputil.BearerToken(r)
            if token != "" {
                if claims, err := jwtManager.ValidateAccessToken(token); err == nil {
                    r = r.WithContext(authz.ContextWithClaims(r.Context(), claims))
                }
            }
            next.ServeHTTP(w, r)
        })
    }
}
```

    1. Create `services/catalog/internal/service/spotlight/adaptive_slice.go`:
       ```go
       // Package spotlight: adaptive slice helper.
       //
       // Implements the HSB-BE-30 1-2-3 layout rule used by 4 multi-item resolvers
       // (personal_pick, latest_news, now_watching, telegram_news). Pulling this out
       // into a generic helper avoids 4 copies of the same len-switch logic and lets
       // each resolver focus on its own data shape.
       package spotlight

       import "math/rand"

       // AdaptiveSlice applies the spotlight 1-2-3 layout rule.
       //   - len(items) == 0 тЖТ returns nil. The resolver MUST treat this as eligibility=false.
       //   - len(items) == 1 тЖТ returns items (single-element slice passed back unchanged).
       //   - len(items) == 2 тЖТ returns a fresh single-element slice holding a randomly
       //     picked element via rng.Intn(2). rng MUST be non-nil; nil rng panics.
       //   - len(items) >= 3 тЖТ returns items[:3] (sub-slice; callers treat as read-only).
       //
       // The N==2 random pick is the part that makes this helper non-trivial тАФ the rng
       // is injected so resolver tests can pin the choice deterministically.
       //
       // Per HSB-BE-30 + workstreams/hero-spotlight/phases/03-dynamic-cards-migration/03-CONTEXT.md
       // `<decisions>` Adaptive 1-2-3 layout section.
       func AdaptiveSlice[T any](items []T, rng *rand.Rand) []T {
           switch len(items) {
           case 0:
               return nil
           case 1:
               return items
           case 2:
               if rng == nil {
                   panic("spotlight.AdaptiveSlice: rng is required when len(items) == 2 (the random-pick branch)")
               }
               idx := rng.Intn(2)
               return []T{items[idx]}
           default:
               return items[:3]
           }
       }
       ```
