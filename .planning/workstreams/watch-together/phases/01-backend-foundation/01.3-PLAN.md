```go
type Hub struct { /* internals */ }

func NewHub(repo *repo.RoomRepo, log *logger.Logger, instanceID string) *Hub
// instanceID = uuid generated at boot; tags every published event so we ignore our own echoes.

// Register adds a fresh Connection to the room set and starts its read/write pumps.
// Returns the Connection so the caller (01.5) can hold a reference for cleanup
// and the inbound handler (01.6) can route on conn.UserID.
func (h *Hub) Register(roomID, userID, username string, conn *websocket.Conn) (*Connection, error)

// Unregister removes a Connection, closes its sendCh, closes the websocket,
// and (if it was the last connection in the room) signals the hub to stop the
// per-room pubsub subscriber goroutine.
func (h *Hub) Unregister(c *Connection)

// Broadcast sends envelope to every member of roomID EXCEPT excludeUserID (pass "" to broadcast to all).
// Also publishes to wt:room:{id}:events (forward-compat). Returns the count of LOCAL recipients
// (those that successfully received via local fanout; pubsub fanout count is not tracked v1.0).
func (h *Hub) Broadcast(ctx context.Context, roomID string, env domain.Envelope, excludeUserID string) (int, error)

// SendTo targets a single user_id in a room. Used for playback:correction and per-recipient errors.
// If the user has multiple connections in the room (tab refresh race), all of them receive.
// Returns the count of connections that received.
func (h *Hub) SendTo(ctx context.Context, roomID, userID string, env domain.Envelope) (int, error)

// MemberUserIDs returns the deduplicated set of currently-connected user IDs for a room.
// O(N) on member count; called on member-list snapshot generation (01.5).
func (h *Hub) MemberUserIDs(roomID string) []string

// MemberCount is the cheap version (raw connection count; multi-tab users counted multiple times).
// Used by capacity-check in 01.5 (capacity is enforced on raw connection count to prevent
// resource abuse via multi-tab spawn).
func (h *Hub) MemberCount(roomID string) int

// Close shuts down all connections and subscriber goroutines (called from main.go on SIGTERM).
func (h *Hub) Close()

type Connection struct {
    UserID   string
    Username string
    RoomID   string
    // sendCh is internal; the hub writes pre-serialized JSON bytes here, the
    // writePump goroutine drains. Buffered size = 64. Slow consumer → channel
    // full → drop the connection (publish lost from this client's perspective
    // but the room continues).
    sendCh chan []byte
    conn   *websocket.Conn
    hub    *Hub
    // Inbound message handler set by 01.6's WS-upgrade wiring. Hub does NOT call this directly —
    // the readPump invokes it. Type:
    //   func(*Connection, domain.Envelope)
    OnMessage func(*Connection, domain.Envelope)
}

// Send writes raw envelope bytes to the connection's send channel (non-blocking; drops on full).
// Returns false if the connection's send buffer was full and the message was dropped.
func (c *Connection) Send(payload []byte) bool
```
