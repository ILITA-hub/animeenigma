    Create `services/gateway/internal/transport/ws_proxy.go`:

    ```go
    package transport

    import (
        "net/http"
        "net/http/httputil"
        "net/url"
        "time"

        "github.com/ILITA-hub/animeenigma/libs/logger"
    )

    // newWSProxy returns an http.HandlerFunc that reverse-proxies WebSocket
    // Upgrade requests to targetBaseURL. Built on httputil.NewSingleHostReverseProxy
    // which handles WS upgrade correctly as of Go 1.12+. Director preserves the
    // request path AND query string so ?token=...&room=... reach the back-end.
    //
    // Why not use the existing ProxyService.Forward? ProxyService deliberately
    // strips RFC 7230 §6.1 hop-by-hop headers including Upgrade + Connection,
    // which is correct for normal HTTP but breaks WS handshake. We use a
    // dedicated handler for WS upgrades; non-WS HTTP traffic stays on the
    // existing ProxyService path.
    func newWSProxy(targetBaseURL string, log *logger.Logger) (http.HandlerFunc, error) {
        target, err := url.Parse(targetBaseURL)
        if err != nil {
            return nil, fmt.Errorf("watch-together ws proxy: parse target: %w", err)
        }
        rp := httputil.NewSingleHostReverseProxy(target)
        originalDirector := rp.Director
        rp.Director = func(req *http.Request) {
            originalDirector(req)
            // Preserve original path verbatim — SingleHostReverseProxy already
            // does this, but be explicit.
            req.Host = target.Host
        }
        rp.FlushInterval = -1 // immediate flushing for streaming WS frames
        rp.ErrorHandler = func(w http.ResponseWriter, r *http.Request, err error) {
            log.Errorw("watch-together ws proxy error", "error", err, "path", r.URL.Path)
            http.Error(w, "bad gateway", http.StatusBadGateway)
        }
        return rp.ServeHTTP, nil
    }
    ```
