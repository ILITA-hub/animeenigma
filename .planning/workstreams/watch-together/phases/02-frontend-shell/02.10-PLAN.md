    ```typescript
    import { test, expect, type BrowserContext, type Page } from '@playwright/test'

    test.describe.configure({ mode: 'serial' })

    const ANIME_ID = '<seed-this-from-ui_audit_bot-seed-data>'  // pick a known anime UUID from the seeded set (check scripts/seed-ui-audit-user.sh for the 8 anime_list entries — use one with a reliable provider, ideally animelib)

    test('two-browser room: create → join → chat → reactions → leave', async ({ browser }) => {
      // ── Setup: two isolated browser contexts so they have separate JWTs ──
      const ctxA = await browser.newContext()
      const ctxB = await browser.newContext()
      const pageA = await ctxA.newPage()
      const pageB = await ctxB.newPage()

      // Both log in (reuse the ui_audit_bot auth flow + a second test account).
      // If only one seeded account exists, register a temp B account via the API.
      await loginAsUiAuditBot(pageA)
      const userB = await registerEphemeralTestUser(pageB)

      // ── A creates a room ──
      await pageA.goto(`/anime/${ANIME_ID}`)
      await pageA.locator('button[aria-label*="watch online" i], button:has-text("Смотреть онлайн"), button:has-text("Watch Now"), button:has-text("Continue")').first().click()  // activate the player
      const inviteBtn = pageA.locator('button:has-text(/Invite to Watch Together|Смотреть вместе/)')
      await expect(inviteBtn).toBeVisible({ timeout: 10_000 })
      await inviteBtn.click()

      // URL should change to /watch/room/<id>
      await pageA.waitForURL(/\/watch\/room\/[a-f0-9-]+/)
      const roomUrl = pageA.url()
      const roomId = roomUrl.match(/\/watch\/room\/([a-f0-9-]+)/)?.[1]
      expect(roomId).toBeTruthy()

      // ── B opens the same room URL ──
      await pageB.goto(roomUrl)
      await pageB.waitForURL(/\/watch\/room\//)

      // ── Both pages should now see each other in MemberList ──
      // Wait for the WS room:snapshot frames to land
      await expect(pageA.locator('aside li')).toHaveCount(2, { timeout: 8_000 })
      await expect(pageB.locator('aside li')).toHaveCount(2, { timeout: 8_000 })

      // ── A sends a chat message; B sees it ──
      const chatTextarea = pageA.locator('textarea[aria-label*="chat" i], textarea[aria-label*="Сообщение" i]')
      await chatTextarea.fill('Hello from A')
      await pageA.keyboard.press('Enter')
      await expect(pageB.locator('ul li:has-text("Hello from A")')).toBeVisible({ timeout: 5_000 })

      // ── B sends a chat back; A sees it ──
      const chatTextareaB = pageB.locator('textarea')
      await chatTextareaB.fill('Hi A')
      await pageB.keyboard.press('Enter')
      await expect(pageA.locator('ul li:has-text("Hi A")')).toBeVisible({ timeout: 5_000 })

      // ── A sends a 🔥 reaction; B sees a burst ──
      await pageA.locator('button[aria-label="🔥"]').click()
      // The burst overlay is inside the player wrapper; assert at least one <span> with the emoji appears within 3s
      await expect(pageB.locator('span:has-text("🔥")').first()).toBeVisible({ timeout: 5_000 })

      // ── B closes their browser context; A sees member:left → MemberList drops to 1 ──
      await ctxB.close()
      await expect(pageA.locator('aside li')).toHaveCount(1, { timeout: 10_000 })

      // ── Cleanup ──
      await ctxA.close()
    })

    test('expired room URL renders room-ended state', async ({ browser }) => {
      const ctx = await browser.newContext()
      const page = await ctx.newPage()
      await loginAsUiAuditBot(page)
      // Navigate to a definitely-non-existent room
      await page.goto('/watch/room/00000000-0000-0000-0000-000000000000')
      // Should render the room-ended title (either en or ru — they have the same key)
      await expect(
        page.locator('text=/This Watch Together room has ended|Эта комната Watch Together завершена/')
      ).toBeVisible({ timeout: 5_000 })
      // And a back button
      await expect(page.locator('button:has-text(/Back to anime|Назад к аниме/)')).toBeVisible()
      await ctx.close()
    })

    // Helpers
    async function loginAsUiAuditBot(page: Page) {
      // Use the API-key approach from CLAUDE.md §UI Audit Test User — fetch a JWT,
      // store in localStorage, then navigate. Mirrors what existing e2e specs do.
      await page.goto('/')
      await page.evaluate(async () => {
        const r = await fetch('/api/auth/login', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          credentials: 'include',
          body: JSON.stringify({ username: 'ui_audit_bot', password: 'audit_bot_test_password_2026' }),
        })
        const data = (await r.json()).data
        localStorage.setItem('token', data.access_token)
        localStorage.setItem('user', JSON.stringify(data.user))
      })
    }

    async function registerEphemeralTestUser(page: Page): Promise<{ username: string }> {
      const username = `wt_test_${Date.now()}_${Math.floor(Math.random() * 1000)}`
      await page.goto('/')
      await page.evaluate(async (u) => {
        const r = await fetch('/api/auth/register', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          credentials: 'include',
          body: JSON.stringify({ username: u, password: 'wt_test_password_long_enough_for_validation' }),
        })
        const data = (await r.json()).data
        localStorage.setItem('token', data.access_token)
        localStorage.setItem('user', JSON.stringify(data.user))
      }, username)
      return { username }
    }
    ```
