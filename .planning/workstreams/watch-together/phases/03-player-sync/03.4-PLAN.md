import { test, expect } from '@playwright/test'

const GATEWAY_HEALTH = '/api/anime/_/scraper/health'  // existing reachable endpoint, same as Phase 2 spec
const PROBE_TIMEOUT_MS = 5000
const COMMAND_RESPONSE_TIMEOUT_MS = 5000

test.beforeAll(async ({ request, baseURL }) => {

test('[canary] kodik_player_api RPC responds to get_time/play/pause/seek', async ({ page }) => {
  // Step 1: navigate to a Kodik-bearing anime. Pick the first ongoing anime
  // (most likely to have Kodik translations). Endpoint already exists for
  // browse listing.
  const browseRes = await page.request.get('/api/anime/browse?status=ongoing&limit=10')
  expect(browseRes.ok()).toBeTruthy()
  const browseBody = await browseRes.json() as { items: Array<{ id: string }> }
  let kodikAnimeId: string | null = null
  for (const item of browseBody.items) {
    const det = await page.request.get(`/api/anime/${item.id}/sources`)
    if (!det.ok()) continue
    const detBody = await det.json() as { sources?: Array<{ type: string }> }
    if (detBody.sources?.some(s => s.type === 'kodik')) {
      kodikAnimeId = item.id
      break
    }
  }
  if (!kodikAnimeId) test.skip(true, 'no Kodik-bearing anime found in first 10 ongoing — catalog drift')

  await page.goto(`/anime/${kodikAnimeId}`)
  await page.getByRole('button', { name: /смотреть|watch/i }).first().click({ timeout: 10000 })
  // Wait for KodikPlayer iframe to render.

  await expect(frame).toBeVisible({ timeout: 15000 })

  // Step 3: install a global event-collector on the parent window so we can
  // assert outbound events without racing.
  await page.evaluate(() => {
    interface WindowWithCanary extends Window {
      __kodikCanary: { events: Array<{ key: string; value: unknown; t: number }> }
    }
    const w = window as unknown as WindowWithCanary
    w.__kodikCanary = { events: [] }
    window.addEventListener('message', (ev) => {
      if (ev?.data && typeof ev.data === 'object' && 'key' in ev.data && typeof (ev.data as { key: unknown }).key === 'string') {
        const k = (ev.data as { key: string }).key
        if (k.startsWith('kodik_player_')) {
          w.__kodikCanary.events.push({ key: k, value: (ev.data as { value: unknown }).value, t: Date.now() })
        }
      }
    })
  })

  // Step 4: send get_time via parent postMessage to the iframe.
  await page.evaluate(() => {
    const ifr = document.querySelector('iframe') as HTMLIFrameElement | null
