```typescript
/**
 * Watch Together — two-browser sync end-to-end (WT-SYNC-09).
 *
 * One test block per player kind. Each test:
 *   1. Creates a room with that player via POST /api/watch-together/rooms.
 *   2. Browser A and Browser B both navigate to invite_url.
 *   3. Browser A drives play/pause/seek.
 *   4. Browser B's state mirrors within 2s (500ms target + slack).
 *
 * Pre-flight gateway probe — spec skips if the dev stack isn't up. Mirrors
 * the Phase 2 watch-together-shell.spec.ts pattern (live-stack only).
 */

import { test, expect, type Browser, type BrowserContext, type Page } from '@playwright/test'

const GATEWAY_HEALTH = '/api/anime/_/scraper/health'
const UI_AUDIT_USER = 'ui_audit_bot'
const UI_AUDIT_PASSWORD = 'audit_bot_test_password_2026'  // per CLAUDE.md "UI Audit Test User"
const SYNC_POLL_TIMEOUT_MS = 2000

interface LoginResult { token: string; userId: string; username: string }

async function login(page: Page, username: string, password: string): Promise<LoginResult> {
  const res = await page.request.post('/api/auth/login', { data: { username, password } })
  if (!res.ok()) throw new Error(`login failed for ${username}: ${res.status()}`)
  const body = await res.json() as { token: string; user: { id: string; username: string } }
  await page.addInitScript(([t, u]) => {
    localStorage.setItem('token', t as string)
    localStorage.setItem('user', JSON.stringify(u))
  }, [body.token, body.user] as const)
  return { token: body.token, userId: body.user.id, username: body.user.username }
}

async function registerEphemeral(page: Page): Promise<LoginResult> {
  const username = `wt_sync_e2e_${Date.now()}`
  const password = 'TestPass!23456789'
  const reg = await page.request.post('/api/auth/register', { data: { username, password } })
  if (!reg.ok()) throw new Error(`register failed: ${reg.status()}`)
  return login(page, username, password)
}

async function resolveAnimeUuid(page: Page, userToken: string): Promise<string> {
  const res = await page.request.get('/api/users/me/anime-list?status=watching', {
    headers: { Authorization: `Bearer ${userToken}` }
  })
  if (!res.ok()) throw new Error('anime-list fetch failed')
  const body = await res.json() as { items?: Array<{ anime?: { id?: string } }> }
  const id = body.items?.[0]?.anime?.id
  if (!id) throw new Error('no seeded anime found — run scripts/seed-ui-audit-user.sh')
  return id
}

async function createRoom(page: Page, token: string, animeId: string, player: string): Promise<{ roomId: string; inviteUrl: string }> {
  const res = await page.request.post('/api/watch-together/rooms', {
    headers: { Authorization: `Bearer ${token}` },
    data: { anime_id: animeId, episode_id: '1', player, translation_id: '' }
  })
  if (!res.ok()) throw new Error(`create room failed: ${res.status()} for player ${player}`)
  const body = await res.json() as { room_id: string; invite_url: string }
  return { roomId: body.room_id, inviteUrl: body.invite_url }
}

test.beforeAll(async ({ request, baseURL }) => {
  if (!baseURL) test.skip(true, 'baseURL not configured')
  try {
    const res = await request.get(`${baseURL}${GATEWAY_HEALTH}`, { timeout: 5000 })
    if (!res.ok()) test.skip(true, `gateway unhealthy — dev stack down?`)
  } catch (err) {
    test.skip(true, `gateway unreachable — ${err}`)
  }
})

// Per-player test suite. Each player has slightly different driver code,
// so we don't use test.describe.parameterize — we write 5 distinct tests
// and share the pair-creation helper.
async function setupPair(browser: Browser, player: string): Promise<{
  ctxA: BrowserContext
  ctxB: BrowserContext
  pageA: Page
  pageB: Page
  inviteUrl: string
}> {
  const ctxA = await browser.newContext()
  const ctxB = await browser.newContext()
  const pageA = await ctxA.newPage()
  const pageB = await ctxB.newPage()
  const authA = await login(pageA, UI_AUDIT_USER, UI_AUDIT_PASSWORD)
  await registerEphemeral(pageB)
  const animeId = await resolveAnimeUuid(pageA, authA.token)
  const { inviteUrl } = await createRoom(pageA, authA.token, animeId, player)
  await pageA.goto(inviteUrl)
  await pageB.goto(inviteUrl)
  return { ctxA, ctxB, pageA, pageB, inviteUrl }
}

// ───────── HTML5 player tests (AnimeLib, OurEnglish, Hanime, Raw) ─────────

for (const player of ['animelib', 'ourenglish', 'hanime', 'raw'] as const) {
  test(`sync: two browsers stay in lockstep on ${player} player`, async ({ browser }) => {
    const { ctxA, ctxB, pageA, pageB } = await setupPair(browser, player)
    try {
      const videoA = pageA.locator('video').first()
      const videoB = pageB.locator('video').first()
      await expect(videoA).toBeVisible({ timeout: 15000 })
      await expect(videoB).toBeVisible({ timeout: 15000 })

      // Play in A → B's video should be playing (not paused) within 2s.
      await videoA.evaluate((v: HTMLVideoElement) => v.play())
      await expect.poll(
        async () => videoB.evaluate((v: HTMLVideoElement) => !v.paused),
        { timeout: SYNC_POLL_TIMEOUT_MS, message: `B did not start playing after A played [${player}]` }
      ).toBe(true)

      // Pause in A → B paused within 2s.
      await videoA.evaluate((v: HTMLVideoElement) => v.pause())
      await expect.poll(
        async () => videoB.evaluate((v: HTMLVideoElement) => v.paused),
        { timeout: SYNC_POLL_TIMEOUT_MS, message: `B did not pause after A paused [${player}]` }
      ).toBe(true)

      // Seek in A to 30s → B currentTime within ±1s of 30s within 2s.
      await videoA.evaluate((v: HTMLVideoElement) => { v.currentTime = 30 })
      await expect.poll(
        async () => videoB.evaluate((v: HTMLVideoElement) => v.currentTime),
        { timeout: SYNC_POLL_TIMEOUT_MS, message: `B did not seek to ~30s after A seeked [${player}]` }
      ).toBeGreaterThan(29)
      const finalTimeB = await videoB.evaluate((v: HTMLVideoElement) => v.currentTime)
      expect(finalTimeB).toBeLessThan(32)
    } finally {
      await ctxA.close()
      await ctxB.close()
    }
  })
}

// ───────── Kodik (iframe RPC) ─────────

test('sync: two browsers stay in lockstep on kodik player', async ({ browser }) => {
  const { ctxA, ctxB, pageA, pageB } = await setupPair(browser, 'kodik')
  try {
    const ifrA = pageA.locator('iframe').first()
    const ifrB = pageB.locator('iframe').first()
    await expect(ifrA).toBeVisible({ timeout: 15000 })
    await expect(ifrB).toBeVisible({ timeout: 15000 })

    // Install event collector on B so we can detect kodik_player_play.
    await pageB.evaluate(() => {
      interface W extends Window { __syncEvents: string[] }
      ;(window as unknown as W).__syncEvents = []
      window.addEventListener('message', (ev) => {
        if (ev?.data?.key && typeof ev.data.key === 'string' && ev.data.key.startsWith('kodik_player_')) {
          ;(window as unknown as W).__syncEvents.push(ev.data.key)
        }
      })
    })

    // Drive play in A via RPC.
    await pageA.evaluate(() => {
      const ifr = document.querySelector('iframe') as HTMLIFrameElement | null
      ifr?.contentWindow?.postMessage({ key: 'kodik_player_api', value: { method: 'play' } }, '*')
    })

    // Assert B's iframe emitted kodik_player_play (driven by the bridge → postCommand('play')).
    await expect.poll(
      async () => pageB.evaluate(() => {
        interface W extends Window { __syncEvents: string[] }
        return (window as unknown as W).__syncEvents.includes('kodik_player_play')
      }),
      { timeout: SYNC_POLL_TIMEOUT_MS, message: 'B did not receive kodik_player_play after A played' }
    ).toBe(true)

    // Skip pause/seek assertions for Kodik if the boot probe failed —
    // detect by looking for the kodik_sync_unavailable banner.
    const fallbackBanner = pageA.getByText(/Kodik sync unavailable|Синхронизация Kodik недоступна/i)
    const fallbackVisible = await fallbackBanner.isVisible().catch(() => false)
    if (fallbackVisible) {
      test.info().annotations.push({ type: 'note', description: 'Kodik boot probe failed — pause/seek assertions skipped' })
      return
    }

    // Drive pause in A → B emits kodik_player_pause.
    await pageB.evaluate(() => {
      interface W extends Window { __syncEvents: string[] }
      ;(window as unknown as W).__syncEvents = []
    })
    await pageA.evaluate(() => {
      const ifr = document.querySelector('iframe') as HTMLIFrameElement | null
      ifr?.contentWindow?.postMessage({ key: 'kodik_player_api', value: { method: 'pause' } }, '*')
    })
    await expect.poll(
      async () => pageB.evaluate(() => {
        interface W extends Window { __syncEvents: string[] }
        return (window as unknown as W).__syncEvents.includes('kodik_player_pause')
      }),
      { timeout: SYNC_POLL_TIMEOUT_MS, message: 'B did not receive kodik_player_pause after A paused' }
    ).toBe(true)
  } finally {
    await ctxA.close()
    await ctxB.close()
  }
})

// ───────── Drift correction under CPU throttling (one player suffices) ─────────

test('sync: drift correction pulls B back when throttled (animelib)', async ({ browser }) => {
  const { ctxA, ctxB, pageA, pageB } = await setupPair(browser, 'animelib')
  try {
    const videoA = pageA.locator('video').first()
    const videoB = pageB.locator('video').first()
    await expect(videoA).toBeVisible({ timeout: 15000 })
    await expect(videoB).toBeVisible({ timeout: 15000 })

    // Throttle B to 4× CPU slowdown.
    const cdp = await pageB.context().newCDPSession(pageB)
    await cdp.send('Emulation.setCPUThrottlingRate', { rate: 4 })

    await videoA.evaluate((v: HTMLVideoElement) => v.play())
    await videoB.evaluate((v: HTMLVideoElement) => v.play())

    // Let drift accumulate (~6s of throttled play).
    await pageA.waitForTimeout(6000)

    // Un-throttle and observe drift-correction handler converge.
    await cdp.send('Emulation.setCPUThrottlingRate', { rate: 1 })
    await pageA.waitForTimeout(3000)

    const [tA, tB] = await Promise.all([
      videoA.evaluate((v: HTMLVideoElement) => v.currentTime),
      videoB.evaluate((v: HTMLVideoElement) => v.currentTime),
    ])
    // Drift should converge to within 3 seconds (soft-correction window is 5s).
    expect(Math.abs(tA - tB)).toBeLessThan(3)
  } finally {
    await ctxA.close()
    await ctxB.close()
  }
})
```
