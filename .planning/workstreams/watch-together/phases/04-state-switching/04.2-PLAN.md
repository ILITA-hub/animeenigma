      // Package service — catalog_client.go is the HTTP back-channel watch-together
      // uses to validate state-change requests against the catalog service.
      // Backs WT-STATE-02; called from inbound.go's handleStateChange (added in
      // Plan 04.3).
      //
      // ...docblock referencing 04-CONTEXT.md §decisions + 04.1 endpoint contract...

      type ValidateResult struct {
          Valid  bool   `json:"valid"`
          Reason string `json:"reason,omitempty"`
      }

      type CatalogClient struct {
          baseURL string
          http    *http.Client
          log     *logger.Logger

          mu    sync.Mutex
          cache map[string]cachedValidation // positive results only
          now   func() time.Time            // injectable for tests
      }

      type cachedValidation struct {
          result   ValidateResult
          expireAt time.Time
      }

      const catalogClientTimeout = 3 * time.Second
      const catalogPositiveCacheTTL = 5 * time.Second

      func NewCatalogClient(baseURL string, log *logger.Logger) *CatalogClient {
          // strip trailing slash from baseURL just in case caller didn't
          baseURL = strings.TrimRight(baseURL, "/")
          return &CatalogClient{
              baseURL: baseURL,
              http:    &http.Client{Timeout: catalogClientTimeout},
              log:     log,
              cache:   make(map[string]cachedValidation),
              now:     time.Now,
          }
      }

      func (c *CatalogClient) ValidateEpisode(ctx context.Context, shikimoriID, player, episodeID, translationID, watchType string) (ValidateResult, error)
